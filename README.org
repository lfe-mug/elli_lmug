# http://stackoverflow.com/q/1404796/1793234#comment1246640_1404862
#+NAME: tag
#+BEGIN_SRC sh :exports none
git describe --tags --abbrev=0
#+END_SRC

Just add it to your =rebar.config= =deps=:

#+BEGIN_SRC erlang
{deps, [
  {lmug_elli,
   {git, "git://github.com/lfe-mug/lmug-elli.git",
    {tag, "<<tag()>>"}}}
]}.
#+END_SRC
